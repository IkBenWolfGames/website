        const int red = 10, green = 11, blue = 9;
        int redpwm, greenpwm, bluepwm;
        
        void setup() {
          pinMode(red, OUTPUT);
          pinMode(green, OUTPUT);
          pinMode(blue, OUTPUT);
        }
        
        void loop() {
          while (redpwm < 255) {               // R-B
            redpwm++;
            analogWrite(red, redpwm);
            delay(5);
          }
        
          while (bluepwm > 0) {                // R
            bluepwm--;
            analogWrite(blue, bluepwm);
            delay(5);
          }
        
          while (greenpwm < 255) {             // R-G
            greenpwm++;
            analogWrite(green, greenpwm);
            delay(5);
          }
        
          while (redpwm > 0) {                 // G
            redpwm--;
            analogWrite(red, redpwm);
            delay(5);
          }
        
          while (bluepwm < 255) {                // G-B
            bluepwm++;
            analogWrite(blue, bluepwm);
            delay(5);
          }
        
          while (greenpwm > 0) {               // B
            greenpwm--;
            analogWrite(green, greenpwm);
            delay(5);
          }
          
        
        }
